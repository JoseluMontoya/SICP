(define (improve guess x)
  (/
   (+ (/ x (square guess))
      (* 2 guess))
   3))

(define (good-enough? guess impr-guess)
  (< (abs (/ (- guess impr-guess) guess)) 0.1))

(define (cube-iter guess x)
  (if (good-enough? guess (improve guess x))
      guess
      (cube-iter (improve guess x) x)))
